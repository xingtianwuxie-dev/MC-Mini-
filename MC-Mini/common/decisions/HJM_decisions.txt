		}

		visible = {
			has_country_flag = HJM_record_13_played
			has_active_mission = HJM_record_13_war_countdown
			NOT = { has_country_flag = HJM_record_13_war_started }
		}

		is_good = no
		selectable_mission = yes
		days_mission_timeout = 180

		cancel_trigger = {
			OR = {
				has_war_with = HIM
				NOT = { HIM = { exists = yes } }
			}
		}

		timeout_effect = {
			set_country_flag = HJM_record_13_war_started
			if = {
				limit = {
					HIM = { exists = yes }
					NOT = { has_war_with = HIM }
				}
				declare_war_on = {
					target = HIM
					type = annex_everything
				}
			}
		}

		complete_effect = {
			custom_effect_tooltip = HJM_record_13_war_countdown_tt
			set_country_flag = HJM_record_13_war_started
			if = {
				limit = {
					HIM = { exists = yes }
					NOT = { has_war_with = HIM }
				}
				declare_war_on = {
					target = HIM
					type = annex_everything
				}
			}
		}
	}

HJM_illager_activity_category = {
	HJM_illager_activity_escalation = {
		icon = GFX_decision_generic_political_pressure
		is_good = no
		selectable_mission = no
		days_mission_timeout = 60

		allowed = {
			original_tag = HJM
		}

		visible = {
			has_country_flag = HJM_illager_activity_unlocked
			has_active_mission = HJM_illager_activity_escalation
			OR = {
				has_idea = HJM_illager_activity_low
				has_idea = HJM_illager_activity_medium
			}
		}

		available = {
			has_country_flag = HJM_illager_activity_unlocked
			has_active_mission = HJM_illager_activity_escalation
			OR = {
				has_idea = HJM_illager_activity_low
				has_idea = HJM_illager_activity_medium
			}
		}

		cancel_trigger = {
			OR = {
				has_idea = HJM_illager_activity_high
				NOT = {
					OR = {
						has_idea = HJM_illager_activity_low
						has_idea = HJM_illager_activity_medium
					}
				}
			}
		}

		timeout_effect = {
			if = {
				limit = {
					has_idea = HJM_illager_activity_low
				}
				remove_ideas = HJM_illager_activity_low
				add_ideas = HJM_illager_activity_medium
				activate_mission = HJM_illager_activity_escalation
			}
			else_if = {
				limit = {
					has_idea = HJM_illager_activity_medium
				}
				remove_ideas = HJM_illager_activity_medium
				add_ideas = HJM_illager_activity_high
			}
		}
	}
