HJM_illager_activity_category = {
	# 清剿灾厄暗线：花 75 政治点压低一级，并重置 60 天升级倒计时
	HJM_reduce_illager_activity = {
		icon = GFX_decision_generic_political_pressure
		cost = 75
		fire_only_once = no
		days_re_enable = 30          # 30 天冷却，防止 AI 高频清剿制造拉锯

		allowed = { original_tag = HJM }

		# 只有 中/高 时才需要清剿（已经是"低"就不用再降）
		visible = {
				has_country_flag = HJM_illager_activity_unlocked
			OR = {
				has_idea = HJM_illager_activity_medium
				has_idea = HJM_illager_activity_high
			}
		}
		available = {
			has_country_flag = HJM_illager_activity_unlocked
			OR = {
				has_idea = HJM_illager_activity_medium
				has_idea = HJM_illager_activity_high
			}
		}

		complete_effect = {
				# 高 → 中
			if = {
				limit = { has_idea = HJM_illager_activity_high }
				remove_ideas = HJM_illager_activity_high
				add_ideas = HJM_illager_activity_medium
			}
			# 中 → 低
			else_if = {
				limit = { has_idea = HJM_illager_activity_medium }
				remove_ideas = HJM_illager_activity_medium
				add_ideas = HJM_illager_activity_low
			}
			# 降级后重启 60 天倒计时（用事件而非 mission，避免闪烁）
			country_event = { id = HJM.illager_tick  days = 60 }
		}

		# AI 只在猖獗度"高"时才清剿，避免 中↔低 反复拉锯
		ai_will_do = {
			factor = 0
			modifier = {
				add = 1
				has_idea = HJM_illager_activity_high
			}
		}
	}
}
