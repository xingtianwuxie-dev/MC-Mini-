on_actions = {
	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					ROOT = { tag = HIM }
				}
				every_country = {
					limit = {
						is_subject_of = HIM
						OR = {
							original_tag = LLL
							original_tag = ZLG
							original_tag = XWY
							original_tag = GWG
						}
						NOT = { has_war_with = FROM }
					}
					add_to_war = {
						targeted_alliance = HIM
						enemy = FROM
						hostility_reason = asked_to_join
					}
				}
			}

			if = {
				limit = {
					FROM = { tag = HIM }
				}
				every_country = {
					limit = {
						is_subject_of = HIM
						OR = {
							original_tag = LLL
							original_tag = ZLG
							original_tag = XWY
							original_tag = GWG
						}
						NOT = { has_war_with = ROOT }
					}
					add_to_war = {
						targeted_alliance = HIM
						enemy = ROOT
						hostility_reason = asked_to_join
					}
				}
			}

			if = {
				limit = {
					ROOT = { tag = HJM }
					FROM = { tag = HIM }
				}
				ROOT = {
					set_country_flag = HJM_record_13_war_activated # Mark the record-13 war chain as active.
					set_country_flag = HJM_record_13_war_started # Mark the HJM-HIM war as started.
					set_country_flag = HJM_war_superevent_start # Show the HJM-HIM war super event to the HJM player.
					set_country_flag = HJM_war_focus_loaded # Prevent later logic from treating the war tree as unloaded.
					load_focus_tree = { tree = HJM_war_focus keep_completed = yes } # Switch HJM into its wartime focus tree.
					add_ideas = HJM_southern_front_crisis # Add the southern front crisis national spirit.
					news_event = { id = HJM_news.2 } # Broadcast the HJM declaration of war news.
					country_event = { id = HJM.illager_purge } # Shut down illager activity when the war begins.
				}
			}

			if = {
				limit = {
					ROOT = { tag = HIM }
					FROM = { tag = HJM }
				}
				FROM = {
					set_country_flag = HJM_record_13_war_activated # Mark the record-13 war chain as active.
					set_country_flag = HJM_record_13_war_started # Mark the HJM-HIM war as started.
					set_country_flag = HJM_war_superevent_start # Show the HJM-HIM war super event to the HJM player.
					set_country_flag = HJM_war_focus_loaded # Prevent later logic from treating the war tree as unloaded.
					load_focus_tree = { tree = HJM_war_focus keep_completed = yes } # Switch HJM into its wartime focus tree.
					add_ideas = HJM_southern_front_crisis # Add the southern front crisis national spirit.
					news_event = { id = HJM_news.2 } # Broadcast the HJM declaration of war news.
					country_event = { id = HJM.illager_purge } # Shut down illager activity when the war begins.
				}
			}

			if = {
				limit = {
					ROOT = {
						tag = HJM
						NOT = { has_global_flag = HJM_MDG_split_happened }
					}
					has_war_with = HIM
				}
				ROOT = {
					country_event = { id = HJM_MDG.1 } # Trigger the End split once when HJM enters the demon war.
				}
			}

			if = {
				limit = {
					FROM = {
						tag = HJM
						NOT = { has_global_flag = HJM_MDG_split_happened }
					}
				}
				FROM = {
					country_event = { id = HJM_MDG.1 } # Trigger the End split once when HJM is the war target.
				}
			}

			if = {
				limit = {
					OR = {
						AND = { ROOT = { tag = HJM } FROM = { tag = HIM } }
						AND = { ROOT = { tag = HIM } FROM = { tag = HJM } }
					}
				}
				every_country = {
					limit = {
						OR = {
							original_tag = HIM
							is_in_faction_with = HIM
						}
					}
					add_ideas = demon_war_fervor # Give the demon camp its wartime fervor spirit.
				}
			}
		}
	}

	on_capitulation = {
		effect = {
			if = {
				limit = {
					ROOT = { tag = HIM }
				}
				every_country = {
					limit = {
						is_subject_of = HIM
						OR = {
							original_tag = LLL
							original_tag = ZLG
							original_tag = XWY
							original_tag = GWG
						}
						has_capitulated = no
					}
					add_ideas = HIM_overlord_collapsed # Subjects suffer shock after HIM collapses.
					add_stability = -0.35 # The overlord collapse destabilizes the subject.
					add_war_support = -0.35 # The overlord collapse breaks subject war support.
					set_country_flag = HIM_subject_overlord_collapsed # Mark the subject as affected by HIM collapse.
				}
			}

			if = {
				limit = {
					ROOT = { tag = HIM }
					FROM = { tag = HJM }
				}
				FROM = {
					news_event = { id = HJM_news.3 } # Broadcast HJM's total victory over HIM.
					if = {
						limit = {
							NOT = { has_country_flag = HJM_him_legacy_unlocked }
						}
						set_country_flag = HJM_him_legacy_unlocked # Unlock the post-HIM legacy cleanup system.
						add_ideas = HJM_him_legacy_unsettled # Add lingering demon-war legacy pressure to HJM.
						country_event = { id = HJM.49 days = 5 } # Show the first report on HIM's postwar remnants.
					}
				}
			}
		}
	}
}
