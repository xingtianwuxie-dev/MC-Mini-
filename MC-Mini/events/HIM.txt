		if = {
			limit = {
				NOT = { has_completed_focus = HJM_brother_returns }
			}
			complete_national_focus = HJM_brother_returns
		}
	}

	option = {
		name = HIM.9.a
		ai_chance = { base = 100 }
		set_country_flag = HJM_record_13_played

			limit = {
				NOT = { has_active_mission = HJM_record_13_war_countdown }
				NOT = { has_country_flag = HJM_record_13_war_activated }
			}
			set_country_flag = HJM_record_13_war_activated
			activate_mission = HJM_record_13_war_countdown
		}
	}
}
