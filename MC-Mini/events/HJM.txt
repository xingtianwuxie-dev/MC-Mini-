# ============ 灾厄村民猖獗度：自动升级（链式事件） ============AI写的！！！！！
# 每 60 天触发一次，按 低→中→高 升一级；到"高"封顶并停止重排。
# 用 country_event 的 days=60 自我重排
country_event = {
    id = HJM.illager_tick
    hidden = yes                    # 内部机制事件，不弹窗
    is_triggered_only = yes

    trigger = {
        tag = HJM
        has_country_flag = HJM_illager_activity_unlocked
    }

    # immediate 在事件触发瞬间执行
    immediate = {
        if = {
            limit = { 
                has_idea = HJM_illager_activity_low 
            }
            remove_ideas = HJM_illager_activity_low
            add_ideas = HJM_illager_activity_medium
        }
        else_if = {
            limit = { 
                has_idea = HJM_illager_activity_medium 
            }
            remove_ideas = HJM_illager_activity_medium
            add_ideas = HJM_illager_activity_high
        }
        # 已是 high：封顶
    }

    option = {
        name = HJM.illager_tick.a
        # 只要还没到"高"，就排下一次升级（60 天后）
        if = {
            limit = { 
                has_country_flag = HJM_illager_activity_unlocked   # 新增：开关被关就不重排
                NOT = { 
                    has_idea = HJM_illager_activity_high 
                } 
            }
            country_event = { 
                id = HJM.illager_tick  days = 60 
            }
        }
    }
}
